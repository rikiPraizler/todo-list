* endpoints
 Books:
 GET api/books - get all books
 GET api/books/{id} - get book by id
 POST api/books - create new book
 PUT api/books/{id} - update book by id
 DELETE api/books/{id} - delete book by id

 Readers:
 GET /api/readers - get all readers
 GET /api/readers/{id} - get reader by id
 POST /api/readers - create new reader
 PUT /api/readers/{id} - update reader by id
 DELETE /api/readers/{id} - delete reader by id

 Loans:
 GET /api/loans – get all loans
 GET /api/loans/book/{bookId} - get loans of book by bookId
 GET /api/loans/reader/{readerId} – get loans of reader by readerId
 POST /api/loans – create laon
 PUT /api/loans/{id}/return - update return loan by id
